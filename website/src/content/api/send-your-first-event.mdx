# Sending your first event

Run this code in your node.js backend.

<CodeWrap client:load>
<div slot="node">
<Code client:load>
```
import Tune from "@tune/sdk";
// Or use this if imports are not supported in your node.js version
// const Tune = require("@tune/sdk");

const ops = new Tune("yourapikey");

const e = {
  name: "Tune!",
}

await ops.events.ingest(e);
```
</Code>
</div>

<div slot="fetch">
<Code client:load>
```
const bearerToken = 'Bearer apikey';
const baseUrl = 'https://api.tune';
const url = `api/${this.version}/log`;
const event = {
  name: "test event"
};

fetch(`${baseUrl}/${url}`, {
  method: 'POST',
  headers: {
    'Authorization': bearerToken,
    'Content-Type': 'application/json'
  },
  body: JSON.stringify(event)
})
.then(response => {
  if (!response.ok) {
    throw new Error('Network response was not ok');
  }
  return response.json();
})
.then(data => {
  console.log('Success:', data);
})
.catch(error => {
  console.error('Error:', error);
});
```
</Code>
</div>
</CodeWrap>
